create table names (
    id int not null auto_increment,
    name varchar(20),
    date_added Date,
    primary key (id)  
);

create table accounts (
    id int not null auto_increment,
    first_name varchar(20),
    last_name varchar(20),
    username varchar(30),
    password varchar(100),
    date_created datetime,
    last_login datetime default null,
    salary decimal(9,2) unsigned,
    age int check (age>=0), 
    primary key (id)
);
