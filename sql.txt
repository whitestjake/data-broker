CREATE DATABASE IF NOT EXISTS dse6100_project1;
USE dse6100_project1;

create table names (
    id int not null auto_increment,
    name varchar(20),
    date_added Date,
    primary key (id)  
);

CREATE TABLE IF NOT EXISTS accounts (
    id int not null auto_increment,
    first_name varchar(20),
    last_name varchar(20),
    username varchar(30),
    password varchar(100),
    date_created datetime,
    salary decimal(9,2) unsigned,
    age int check (age>=0),
    registration_date DATETIME DEFAULT CURRENT_TIMESTAMP,
    last_login_date DATETIME NULL -- Save data after user login,
    primary key (id)
);
